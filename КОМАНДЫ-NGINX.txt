╔══════════════════════════════════════════════════════════════╗
║              📋 КОМАНДЫ ДЛЯ ПОИСКА NGINX                    ║
╚══════════════════════════════════════════════════════════════╝

Скопируйте и выполните эти команды на сервере:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣  ПРОВЕРКА NGINX:

    nginx -v

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2️⃣  НАЙТИ КОНФИГУРАЦИЮ DIRECTUS:

    grep -r "cycloscope" /etc/nginx/

    # Или по порту:
    grep -r "8055" /etc/nginx/

    # Или по домену:
    grep -r "morent-sochi" /etc/nginx/

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3️⃣  НАЙТИ НАСТРОЙКИ CORS:

    grep -r "Access-Control" /etc/nginx/

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

4️⃣  ПОСМОТРЕТЬ ФАЙЛ (замените путь на найденный):

    sudo cat /etc/nginx/sites-available/directus

    # Или
    sudo cat /etc/nginx/conf.d/directus.conf

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

5️⃣  РЕДАКТИРОВАТЬ ФАЙЛ:

    sudo nano /etc/nginx/sites-available/directus

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

6️⃣  ПРОВЕРИТЬ КОНФИГУРАЦИЮ:

    sudo nginx -t

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

7️⃣  ПЕРЕЗАГРУЗИТЬ NGINX:

    sudo systemctl reload nginx

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📝 ЧТО ИСКАТЬ В ФАЙЛЕ:

Ищите строки типа:

    add_header 'Access-Control-Allow-Origin' '...' always;

Или:

    set $cors_origin "";
    if ($http_origin ~* (...)) {
        set $cors_origin $http_origin;
    }

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✏️ ЧТО ИЗМЕНИТЬ:

Добавьте dashboardmorent.vercel.app к существующим доменам.

Было:
    'https://morent-sochi.vercel.app'

Стало:
    'https://morent-sochi.vercel.app, https://dashboardmorent.vercel.app'

Или в regex:
    (https://morent-sochi\.vercel\.app|https://dashboardmorent\.vercel\.app)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

💾 РЕЗЕРВНАЯ КОПИЯ:

Перед изменением:

    sudo cp /etc/nginx/sites-available/directus /etc/nginx/sites-available/directus.backup

Восстановить если что-то пошло не так:

    sudo cp /etc/nginx/sites-available/directus.backup /etc/nginx/sites-available/directus
    sudo systemctl reload nginx
